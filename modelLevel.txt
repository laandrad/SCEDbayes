
  model {
          for(i in 1:nPoints){
            y[i] ~ dnorm(mu[i], tau)
            mu[i] <- beta0 + beta1 * P1[i] + beta2 * P2[i] + beta3 * P3[i]
          }

          beta0 ~ dnorm(0, 0.001)
          beta1 ~ dnorm(0, 0.001)
          beta2 ~ dnorm(0, 0.001)
          beta3 ~ dnorm(0, 0.001)
          tau ~ dgamma(0.001, 0.001)
        }
